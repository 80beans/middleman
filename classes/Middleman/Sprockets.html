    def render_path(path)
      source = File.join(options.public, path)
      if File.extname(path) == '.js' && File.exists?(source)
        secretary = ::Sprockets::Secretary.new( :asset_root   => options.public,
                                                :source_files => [source] )
        secretary.concatenation.to_s
      else
        super
      end
    end
