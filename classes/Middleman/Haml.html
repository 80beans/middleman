    def render_path(path)
      if template_exists?(path, :haml)
        result = nil
        begin
          result = haml(path.to_sym, :layout => File.extname(path) != ".xml")
        rescue ::Haml::Error => e
          result = "Haml Error: #{e}"
          result << "<pre>Backtrace: #{e.backtrace.join("\n")}</pre>"
        end
        result
      else
        super
      end
    end
